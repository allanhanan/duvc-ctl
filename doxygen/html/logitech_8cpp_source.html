62Result<bool> supports_logitech_properties(const Device &device) {
63 try {
64 KsPropertySet prop_set(device);
65 if (!prop_set.is_valid()) {
66 return Ok(false);
67 }
68
69 // Try to query support for a basic property
70 auto result = prop_set.query_support(
71 LOGITECH_PROPERTY_SET,
72 static_cast<uint32_t>(LogitechProperty::RightLight));
73
74 if (result.is_ok()) {
75 uint32_t support_flags = result.value();
76 return Ok((support_flags &
77 (KSPROPERTY_SUPPORT_GET | KSPROPERTY_SUPPORT_SET)) != 0);
78 }
79
80 return Ok(false);
81
82 } catch (const std::exception &e) {
83 DUVC_LOG_DEBUG("Exception checking Logitech support: " +
84 std::string(e.what()));
85 return Ok(false); // Assume not supported on error
86 }

89template <typename T>
90Result<T> get_logitech_property_typed(const Device &device,
91 LogitechProperty prop) {
92 auto data_result = get_logitech_property(device, prop);
93 if (!data_result.is_ok()) {
94 return Err<T>(data_result.error());
95 }
96
97 const auto &data = data_result.value();
98 if (data.size() != sizeof(T)) {
99 return Err<T>(ErrorCode::InvalidValue,
100 "Property data size mismatch for Logitech property");
101 }
102
103 T value;
104 std::memcpy(&value, data.data(), sizeof(T));
105 return Ok(value);
106}
