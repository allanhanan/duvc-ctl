76std::wstring read_device_path(IMoniker *mon) {
77 com_ptr<IPropertyBag> bag;
78 if (SUCCEEDED(mon->BindToStorage(nullptr, nullptr, IID_IPropertyBag,
79 reinterpret_cast<void **>(bag.put())))) {
80 auto dp = read_prop_bstr(bag.get(), L"DevicePath");
81 if (!dp.empty())
82 return dp;

88 if (SUCCEEDED(mon->GetDisplayName(nullptr, nullptr, &disp)) && disp) {

90 CoTaskMemFree(disp);

96bool is_same_device(const Device &d, const std::wstring &name,
97 const std::wstring &path) {
98 if (!d.path.empty() && !path.empty()) {
99 if (_wcsicmp(d.path.c_str(), path.c_str()) == 0)
100 return true;
101 }
102
103 if (!d.name.empty() && !name.empty()) {
104 if (_wcsicmp(d.name.c_str(), name.c_str()) == 0)
105 return true;
106 }
107
108 return false;
109}

133bool is_device_connected(const Device &dev) {
134 try {
135 // First try: Check if device still exists in enumeration
136 com_apartment com;
137 auto de = create_dev_enum();
138 auto en = enum_video_devices(de.get());
139 if (!en)
140 return false;
141
142 ULONG fetched = 0;
143 com_ptr<IMoniker> mon;
144 while (en->Next(1, mon.put(), &fetched) == S_OK && fetched) {
145 auto fname = read_friendly_name(mon.get());
146 auto dpath = read_device_path(mon.get());
147 if (is_same_device(dev, fname, dpath)) {
148 // Found in enumeration - now try lightweight access test
149 try {
150 DeviceConnection conn(dev);
151 return conn.is_valid();
152 } catch (...) {
153 // If connection fails, device exists but might be busy
154 return true;
155 }
156 }
157 mon.reset();
158 }
159
160 return false; // Not found in enumeration
161 } catch (...) {
162 return false;
163 }
164}
