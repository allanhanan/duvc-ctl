30static LRESULT CALLBACK device_notification_wndproc(HWND hwnd, UINT msg,
31 WPARAM wParam,
32 LPARAM lParam) {
33 if (msg == WM_DEVICECHANGE && g_device_callback) {
34 DUVC_LOG_DEBUG("Received device change notification");
35
36 if (wParam == DBT_DEVICEARRIVAL || wParam == DBT_DEVICEREMOVECOMPLETE) {
37 PDEV_BROADCAST_HDR hdr = reinterpret_cast<PDEV_BROADCAST_HDR>(lParam);
38
39 if (hdr && hdr->dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE) {
40 PDEV_BROADCAST_DEVICEINTERFACE dev_iface =
41 reinterpret_cast<PDEV_BROADCAST_DEVICEINTERFACE>(lParam);
42
43 bool device_added = (wParam == DBT_DEVICEARRIVAL);
44 std::wstring device_path = dev_iface->dbcc_name;
45
46 DUVC_LOG_INFO(std::string("Device ") +
47 (device_added ? "added: " : "removed: ") +
48 std::string(device_path.begin(), device_path.end()));
49
50 // Call user callback
51 try {
52 g_device_callback(device_added, device_path);
53 } catch (const std::exception &e) {
54 DUVC_LOG_ERROR("Exception in device change callback: " +
55 std::string(e.what()));
56 } catch (...) {
57 DUVC_LOG_ERROR("Unknown exception in device change callback");
58 }
59 }
60 }
61 }
62
63 return DefWindowProc(hwnd, msg, wParam, lParam);
64}

141void register_device_change_callback(DeviceChangeCallback callback) {
142 // Don't register multiple times
143 if (g_notification_window) {
144 DUVC_LOG_WARNING("Device change callback already registered");
145 return;
146 }
147
148 g_device_callback = callback;
149
150 // Create invisible window for receiving notifications
151 g_notification_window = create_notification_window();
152 if (!g_notification_window) {
153 g_device_callback = nullptr;
154 return;
155 }
156
157 // Register for device interface notifications
158 g_device_notify = register_device_notifications(g_notification_window);
159 if (!g_device_notify) {
160 DestroyWindow(g_notification_window);
161 g_notification_window = nullptr;
162 g_device_callback = nullptr;
163 return;
164 }
165
166 DUVC_LOG_INFO("Device change monitoring started");
167}

180 }
181
182 g_device_callback = nullptr;
183 DUVC_LOG_INFO("Device change monitoring stopped");
184}
