23KsPropertySet::KsPropertySet(const Device &device) {
24 try {
25 auto filter = open_device_filter(device);
26 if (filter) {
27 // Try to get IKsPropertySet interface
28 IKsPropertySet *props = nullptr;
29 HRESULT hr = filter->QueryInterface(IID_IKsPropertySet,
30 reinterpret_cast<void **>(&props));
31 if (SUCCEEDED(hr) && props) {
32 property_set_ = com_ptr<IKsPropertySet>(props);
33 }
34 }
35 } catch (...) {
36 // Constructor should not throw
37 property_set_.reset();
38 }
39}

50Result<uint32_t> KsPropertySet::query_support(const GUID &property_set,
51 uint32_t property_id) {
52 if (!is_valid()) {
53 return Err<uint32_t>(ErrorCode::SystemError,
54 "Property set interface not available");
55 }
56
57 ULONG type_support = 0;
58 HRESULT hr =
59 property_set_->QuerySupported(property_set, property_id, &type_support);
60
61 if (FAILED(hr)) {
62 return Err<uint32_t>(ErrorCode::PropertyNotSupported,
63 "Property not supported: " + decode_hresult(hr));
64 }
65
66 return Ok(static_cast<uint32_t>(type_support));
67}

69Result<std::vector<uint8_t>>
70KsPropertySet::get_property(const GUID &property_set, uint32_t property_id) {
71 if (!is_valid()) {
72 return Err<std::vector<uint8_t>>(ErrorCode::SystemError,
73 "Property set interface not available");
74 }
75
76 // First, get the required buffer size
77 ULONG bytes_returned = 0;
78 HRESULT hr = property_set_->Get(property_set, property_id, nullptr, 0,
79 nullptr, 0, &bytes_returned);
80
81 if (FAILED(hr) || bytes_returned == 0) {
82 return Err<std::vector<uint8_t>>(ErrorCode::PropertyNotSupported,
83 "Failed to get property size: " +
84 decode_hresult(hr));
85 }
86
87 // Allocate buffer and get the actual data
88 std::vector<uint8_t> data(bytes_returned);
89 hr = property_set_->Get(property_set, property_id, nullptr, 0, data.data(),
90 bytes_returned, &bytes_returned);
91
92 if (FAILED(hr)) {
93 return Err<std::vector<uint8_t>>(ErrorCode::SystemError,
94 "Failed to get property data: " +
95 decode_hresult(hr));
96 }
97
98 // Resize to actual returned size
99 data.resize(bytes_returned);
100 return Ok(std::move(data));
101}

123template <typename T>
124Result<T> KsPropertySet::get_property_typed(const GUID &property_set,
125 uint32_t property_id) {
126 auto result = get_property(property_set, property_id);
127 if (!result.is_ok()) {
128 return Err<T>(result.error());
129 }
130
131 const auto &data = result.value();
132 if (data.size() != sizeof(T)) {
133 return Err<T>(ErrorCode::InvalidValue,
134 "Property data size mismatch: expected " +
135 std::to_string(sizeof(T)) + " bytes, got " +
136 std::to_string(data.size()));
137 }
138
139 T value;
140 std::memcpy(&value, data.data(), sizeof(T));
141 return Ok(value);
142}
